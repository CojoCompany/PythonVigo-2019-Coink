def read_registers(i2c):

    read_address = 0x6b
    trigger = 1 << 7
    request = bytearray([read_address, trigger])
    reply = bytearray(7)

    i2c.start()
    ack = i2c.write(request)
    i2c.readinto(reply)
    i2c.stop()

    return reply, ack
